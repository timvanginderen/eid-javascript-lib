311 /**
312  * Parses text from the given string to produce a date.
313  * @public
314  * @method parse
315  * @param dateString a string to be parsed.
316  * @throws NullPointerException if dateString is null or undefined.
317  * @throws IllegalArgumentException if dateString does not contain a valid date.
318  * @return a Javascript Date object
319  * @type Date
320  */
321 be.belgium.eid.DateFormatter.prototype.parse = function(dateString) {
322 	if (dateString === null || typeof(dateString) == "undefined")
323 		throw new be.belgium.eid.NullPointerException();
324 
325 	var day = 1;
326 	var month = 1;
327 	var year = 1970;
328 
329 	if (this.dateFormat == be.belgium.eid.dateFormat.EID_BIRTH_DATE) {
330 		var length = dateString.length;
331 		var regExp;
332 		if (length < 11 || length > 12) // format DD mmmm YYYY (Dutch, French) or DD.MMM.YYYY (German)
333 			throw new be.belgium.eid.IllegalArgumentException();
334 		day = dateString.substr(0, 2);
335 		for (var i = 0; i < 12; i++) {
336 			regExp = new RegExp(this.eIDBirthDateAbbreviations[i].join("|"), "i");
337 			if (regExp.test(dateString)) {
338 				month = (i + 1);
339 				break;
340 			}
341 		}
342 		year = dateString.substr((length - 4), 4);
343 	} else {
344 		switch (this.dateFormat) {
345 			case be.belgium.eid.dateFormat.DDMMYYYY :
346 				if (dateString.length != 8)
347 					throw new be.belgium.eid.IllegalArgumentException();
348 				day = dateString.substr(0, 2);
349 				month = dateString.substr(2, 2);
350 				year = dateString.substr(4, 4);
351 				break;
352 			case be.belgium.eid.dateFormat.MMDDYYYY :
353 				if (dateString.length != 8)
354 					throw new be.belgium.eid.IllegalArgumentException();
355 				month = dateString.substr(0, 2);
356 				day = dateString.substr(2, 2);
357 				year = dateString.substr(4, 4);
358 				break;
359 			case be.belgium.eid.dateFormat.YYYYMMDD :
360 				if (dateString.length != 8)
361 					throw new be.belgium.eid.IllegalArgumentException();
362 				year = dateString.substr(0, 4);
363 				month = dateString.substr(4, 2);
364 				day = dateString.substr(6, 2);
365 				break;
366 			case be.belgium.eid.dateFormat.MM_DD_YYYY :
367 				if (dateString.length != 10)
368 					throw new be.belgium.eid.IllegalArgumentException();
369 				month = dateString.substr(0, 2);
370 				day = dateString.substr(3, 2);
371 				year = dateString.substr(6, 4);
372 				break;
373 			case be.belgium.eid.dateFormat.YYYY_MM_DD :
374 				if (dateString.length != 10)
375 					throw new be.belgium.eid.IllegalArgumentException();
376 				year = dateString.substr(0, 4);
377 				month = dateString.substr(5, 2);
378 				day = dateString.substr(8, 2);
379 				break;
380 			default: // be.belgium.eid.dateFormat.DD_MM_YYYY
381 				if (dateString.length != 10)
382 					throw new be.belgium.eid.IllegalArgumentException();
383 				day = dateString.substr(0, 2);
384 				month = dateString.substr(3, 2);
385 				year = dateString.substr(6, 4);
386 				break;
387 		}
388 	}
389 
390 	if (day === "  ") day = 1;
391 	if (month === "  ") month = 1;
392 
393 	return new Date(year, (month - 1), day, 0, 0, 0, 0);
394 };

396 /**
397  * formats a given date into a date string.
398  * @public
399  * @method format
400  * @param date a date to be formatted.
401  * @throws NullPointerException if date is null or undefined.
402  * @return a formatted date string
403  * @type primitive string 
404  */
405 be.belgium.eid.DateFormatter.prototype.format = function(date) {
406 	if (date === null || typeof(date) == "undefined")
407 		throw new be.belgium.eid.NullPointerException();
408 
409 	var returnValue = "";
410 
411 	var day = date.getDate();
412 	if (day < 10) day = "0" + day; // zero padding
413 
414 	var month = date.getMonth();
415 	if (this.dateFormat == be.belgium.eid.dateFormat.EID_BIRTH_DATE) {
416 		var nbrAbbrev = this.eIDBirthDateAbbreviations[month].length;
417 		if (this.language < 0 || this.language >= nbrAbbrev) {
418 			month = this.eIDBirthDateAbbreviations[month][0];
419 		} else {
420 			month = this.eIDBirthDateAbbreviations[month][this.language];
421 		}
422 	} else {
423 		month = month + 1;
424 		if (month < 10) month = "0" + month; // zero padding
425 	}
426 
427 	var year = date.getFullYear();
428 
429 	switch (this.dateFormat) {
430 		case be.belgium.eid.dateFormat.DDMMYYYY :
431 			returnValue = "" + day + "" + month + "" + year;
432 			break;
433 		case be.belgium.eid.dateFormat.MMDDYYYY :
434 			returnValue = "" + month + "" + day + "" + year;
435 			break;
436 		case be.belgium.eid.dateFormat.YYYYMMDD :
437 			returnValue = "" + year + "" + month + "" + day;
438 			break;
439 		case be.belgium.eid.dateFormat.MM_DD_YYYY :
440 			returnValue = "" + month + this.dateSeparator + day + this.dateSeparator + year;
441 			break;
442 		case be.belgium.eid.dateFormat.YYYY_MM_DD :
443 			returnValue = "" + year + this.dateSeparator + month + this.dateSeparator + day;
444 			break;
445 		case be.belgium.eid.dateFormat.EID_BIRTH_DATE :
446 			returnValue = "" + day + this.dateSeparator + month + this.dateSeparator + year;
447 			break;
448 		default: // be.belgium.eid.dateFormat.DD_MM_YYYY
449 			returnValue = "" + day + this.dateSeparator + month + this.dateSeparator + year;
450 			break;
451 	}
452 	return returnValue;
453 };
