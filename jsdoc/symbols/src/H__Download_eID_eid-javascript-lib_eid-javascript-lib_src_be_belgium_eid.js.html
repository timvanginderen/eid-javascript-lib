335 /**
336  * Parses text from the given string to produce a date.
337  * @public
338  * @method parse
339  * @param dateString a string to be parsed.
340  * @throws NullPointerException if dateString is null or undefined.
341  * @throws IllegalArgumentException if dateString does not contain a valid date.
342  * @return a Javascript Date object
343  * @type Date
344  */
345 be.belgium.eid.DateFormatter.prototype.parse = function(dateString) {
346 	if (dateString === null || typeof(dateString) == "undefined")
347 		throw new be.belgium.eid.NullPointerException();
348 
349 	var day = 1;
350 	var month = 1;
351 	var year = 1970;
352 
353 	if (this.dateFormat == be.belgium.eid.dateFormat.EID_BIRTH_DATE) {
354 		var length = dateString.length;
355 		var regExp;
356 		if (length < 11 || length > 12) // format DD mmmm YYYY (Dutch, French) or DD.MMM.YYYY (German)
357 			throw new be.belgium.eid.IllegalArgumentException();
358 		day = dateString.substr(0, 2);
359 		for (var i = 0; i < 12; i++) {
360 			regExp = new RegExp(this.eIDBirthDateAbbreviations[i].join("|"), "i");
361 			if (regExp.test(dateString)) {
362 				month = (i + 1);
363 				break;
364 			}
365 		}
366 		year = dateString.substr((length - 4), 4);
367 	} else {
368 		switch (this.dateFormat) {
369 			case be.belgium.eid.dateFormat.DDMMYYYY :
370 				if (dateString.length != 8)
371 					throw new be.belgium.eid.IllegalArgumentException();
372 				day = dateString.substr(0, 2);
373 				month = dateString.substr(2, 2);
374 				year = dateString.substr(4, 4);
375 				break;
376 			case be.belgium.eid.dateFormat.MMDDYYYY :
377 				if (dateString.length != 8)
378 					throw new be.belgium.eid.IllegalArgumentException();
379 				month = dateString.substr(0, 2);
380 				day = dateString.substr(2, 2);
381 				year = dateString.substr(4, 4);
382 				break;
383 			case be.belgium.eid.dateFormat.YYYYMMDD :
384 				if (dateString.length != 8)
385 					throw new be.belgium.eid.IllegalArgumentException();
386 				year = dateString.substr(0, 4);
387 				month = dateString.substr(4, 2);
388 				day = dateString.substr(6, 2);
389 				break;
390 			case be.belgium.eid.dateFormat.MM_DD_YYYY :
391 				if (dateString.length != 10)
392 					throw new be.belgium.eid.IllegalArgumentException();
393 				month = dateString.substr(0, 2);
394 				day = dateString.substr(3, 2);
395 				year = dateString.substr(6, 4);
396 				break;
397 			case be.belgium.eid.dateFormat.YYYY_MM_DD :
398 				if (dateString.length != 10)
399 					throw new be.belgium.eid.IllegalArgumentException();
400 				year = dateString.substr(0, 4);
401 				month = dateString.substr(5, 2);
402 				day = dateString.substr(8, 2);
403 				break;
404 			default: // be.belgium.eid.dateFormat.DD_MM_YYYY
405 				if (dateString.length != 10)
406 					throw new be.belgium.eid.IllegalArgumentException();
407 				day = dateString.substr(0, 2);
408 				month = dateString.substr(3, 2);
409 				year = dateString.substr(6, 4);
410 				break;
411 		}
412 	}
413 
414 	if (day === "  ") day = 1;
415 	if (month === "  ") month = 1;
416 
417 	return new Date(year, (month - 1), day, 0, 0, 0, 0);
418 };

420 /**
421  * formats a given date into a date string.
422  * @public
423  * @method format
424  * @param date a date to be formatted.
425  * @throws NullPointerException if date is null or undefined.
426  * @return a formatted date string
427  * @type primitive string 
428  */
429 be.belgium.eid.DateFormatter.prototype.format = function(date) {
430 	if (date === null || typeof(date) == "undefined")
431 		throw new be.belgium.eid.NullPointerException();
432 
433 	var returnValue = "";
434 
435 	var day = date.getDate();
436 	if (day < 10) day = "0" + day; // zero padding
437 
438 	var month = date.getMonth();
439 	if (this.dateFormat == be.belgium.eid.dateFormat.EID_BIRTH_DATE) {
440 		var nbrAbbrev = this.eIDBirthDateAbbreviations[month].length;
441 		if (this.language < 0 || this.language >= nbrAbbrev) {
442 			month = this.eIDBirthDateAbbreviations[month][0];
443 		} else {
444 			month = this.eIDBirthDateAbbreviations[month][this.language];
445 		}
446 	} else {
447 		month = month + 1;
448 		if (month < 10) month = "0" + month; // zero padding
449 	}
450 
451 	var year = date.getFullYear();
452 
453 	switch (this.dateFormat) {
454 		case be.belgium.eid.dateFormat.DDMMYYYY :
455 			returnValue = "" + day + "" + month + "" + year;
456 			break;
457 		case be.belgium.eid.dateFormat.MMDDYYYY :
458 			returnValue = "" + month + "" + day + "" + year;
459 			break;
460 		case be.belgium.eid.dateFormat.YYYYMMDD :
461 			returnValue = "" + year + "" + month + "" + day;
462 			break;
463 		case be.belgium.eid.dateFormat.MM_DD_YYYY :
464 			returnValue = "" + month + this.dateSeparator + day + this.dateSeparator + year;
465 			break;
466 		case be.belgium.eid.dateFormat.YYYY_MM_DD :
467 			returnValue = "" + year + this.dateSeparator + month + this.dateSeparator + day;
468 			break;
469 		case be.belgium.eid.dateFormat.EID_BIRTH_DATE :
470 			returnValue = "" + day + this.dateSeparator + month + this.dateSeparator + year;
471 			break;
472 		default: // be.belgium.eid.dateFormat.DD_MM_YYYY
473 			returnValue = "" + day + this.dateSeparator + month + this.dateSeparator + year;
474 			break;
475 	}
476 	return returnValue;
477 };

1941  * Read eID card or SIS card
1942  * @public
1943  * @method read
1944  * @return a EIDCard or SISCard object or null if no card is present or there was a failure to read the card.
1945  * @type EIDCard,SISCard,null
1946  */
1947 be.belgium.eid.CardReader.prototype.read = function() {
1948 	var card = null;
1949 	var cardBuilder = null;
1950 
1951 	try {
1952 		// Reset all objects and previous read card values
1953 		this.getBEIDApplet().InitLib(null);
1954 
1955 		this.determineReaderName();
1956 
1957 		// Still no reader name ...
1958 		if (this.readerName === "") {
1959 			if (this.noReaderDetectedHandler)
1960 				this.noReaderDetectedHandler();
1961 
1962 			try {
1963 				this.getBEIDApplet().exitLib();
1964 			} catch (e){}
1965 
1966 			return null;
1967 		}
1968 
1969 		this.getBEIDApplet().InitLib(this.readerName);
1970 
1971 		if (this.BEIDApplet.isCardPresent(this.readerName)) {
1972 			if (be.belgium.eid.CardReader.validChipNumber(this.BEIDApplet.getChipNumber())) {
1973 				cardBuilder = new be.belgium.eid.EIDCardBuilder35(this.BEIDApplet.getCardNumber());
1974 				cardBuilder.setChipNumber(this.BEIDApplet.getChipNumber());
1975 				cardBuilder.setValidityDateBegin(this.BEIDApplet.getValidityDateBegin());
1976 				cardBuilder.setValidityDateEnd(this.BEIDApplet.getValidityDateEnd());
1977 				cardBuilder.setIssMunicipality(this.BEIDApplet.getIssMunicipality());
1978 				cardBuilder.setNationalNumber(this.BEIDApplet.getNationalNumber());
1979 				cardBuilder.setSurname(this.BEIDApplet.getSurname());
1980 				cardBuilder.setFirstName(this.BEIDApplet.getFirstName());
1981 				cardBuilder.setNationality(this.BEIDApplet.getNationality());
1982 				cardBuilder.setBirthLocation(this.BEIDApplet.getBirthLocation());
1983 				cardBuilder.setBirthDate(this.BEIDApplet.getBirthDate());
1984 				cardBuilder.setSex(this.BEIDApplet.getSex());
1985 				cardBuilder.setNobleCondition(this.BEIDApplet.getNobleCondition());
1986 				cardBuilder.setSpecialStatus(this.BEIDApplet.getSpecialStatus());
1987 				cardBuilder.setStreet(this.BEIDApplet.getStreet());
1988 				cardBuilder.setStreetNumber(this.BEIDApplet.getStreetNumber());
1989 				cardBuilder.setBoxNumber(this.BEIDApplet.getBoxNumber());
1990 				cardBuilder.setZip(this.BEIDApplet.getZip());
1991 				cardBuilder.setMunicipality(this.BEIDApplet.getMunicipality());
1992 				cardBuilder.setCountry(this.BEIDApplet.getCountry());
1993 				cardBuilder.setPicture(this.BEIDApplet.GetPicture());
1994 			} else { 
1995 				// The applet does not return a chip number for SIS cards.
1996 				cardBuilder = new be.belgium.eid.SISCardBuilder35();
1997 				cardBuilder.setCardNumber(this.BEIDApplet.getCardNumber());
1998 				cardBuilder.setValidityDateBegin(this.BEIDApplet.getValidityDateBegin());
1999 				cardBuilder.setValidityDateEnd(this.BEIDApplet.getValidityDateEnd());
2000 				cardBuilder.setNationalNumber(this.BEIDApplet.getNationalNumber());
2001 				cardBuilder.setSurname(this.BEIDApplet.getSurname());
2002 				cardBuilder.setFirstName(this.BEIDApplet.getFirstName());
2003 				cardBuilder.setInitials(this.BEIDApplet.getInitials());
2004 				cardBuilder.setBirthDate(this.BEIDApplet.getBirthDate());
2005 				cardBuilder.setSex(this.BEIDApplet.getSex());
2006 			}
2007 
2008 			card = cardBuilder.getCard();
2009 
2010 		} else {
2011 			if (this.noCardPresentHandler)
2012 				this.noCardPresentHandler();
2013 		}
2014 	} catch (e) {
2015 		if (e instanceof be.belgium.eid.NullPointerException) {
2016 			if (this.appletNotFoundHandler) {
2017 				this.appletNotFoundHandler();
2018 			} else {
2019 				window.alert("BEID Applet not found.");
2020 			}
2021 		} else {
2022 			if (this.appletExceptionHandler) {
2023 				if (this.appletExceptionHandler.length > 0)
2024 					this.appletExceptionHandler(e);
2025 				else
2026 					this.appletExceptionHandler();
2027 			} else {
2028 				window.alert("BEID Applet throw exception: " + e);
2029 			}
2030 		}
2031 	}
2032 
2033 	try {
2034 		this.getBEIDApplet().exitLib();
2035 	} catch (e){}
2036 
2037 	return card;
2038 };
